\section{Expressions}

    \subsection{Expressions with numbers}

    These rules describe expressions in which numbers appear as operands. The other
    operand may be a list. The language
    defines the operators in these expressos as the simpler ones.

    \begin{bnf*}
        \bnfprod{expr}{\bnfpn{term} \bnfor \bnfpn{term}\bnfpn{expr-aux}}\\
        \bnfprod{expr-aux}{\bnfpn{bin-op-p0}\bnfpn{term} \bnfor \bnfpn{bin-op-p0}\bnfpn{term}\bnfpn{expr-aux}}\\
        \bnfprod{term}{\bnfpn{un-op} \bnfpn{term} \bnfor \bnfpn{un-op} \bnfpn{term} \bnfpn{term-aux} \bnfor \bnfpn{factor} \bnfor \bnfpn{factor} \bnfpn{term-aux}}\\
        \bnfprod{term-aux}{\bnfpn{bin-op-p1}\bnfpn{factor} \bnfor \bnfpn{bin-op-p1}\bnfpn{factor}\bnfpn{term-aux}}\\
        \bnfprod{un-op}{\bnfts{+} \bnfor \bnfts{-}}\\
        \bnfprod{bin-op-p0}{\bnfts{+} \bnfor \bnfts{-}}\\
        \bnfprod{bin-op-p1}{\bnfts{*} \bnfor \bnfts{/} \bnfor \bnfts{\%} \bnfor \bnfts{\^}}\\
        \bnfprod{factor}{\bnfts{(} \bnfsp \bnfpn{expr} \bnfsp \bnfts{)} \bnfor \bnfpn{ident} \bnfor \bnfpn{int-lit} \bnfor \bnfpn{float-lit}
        \bnfor \bnfpn{list-lit} \bnfor \bnfpn{subprog-call} }
    \end{bnf*}
